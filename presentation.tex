\section{Conclusions}

\begin{frame}{Key Findings}
  \begin{enumerate}
    \item \textbf{Legendary Dominance:} 44\% higher stats than non-legendary
    \item \textbf{Power Creep:} Modest but visible across generations
    \item \textbf{Type Hierarchy:} Dragon, Steel, Psychic consistently strong
    \item \textbf{Predictability:} 95\% accuracy in legendary classification
    \item \textbf{Physical Correlations:} Size relates to bulk, not speed
    \item \textbf{Archetypes:} Most Pokémon are balanced; extremes are rare
  \end{enumerate}
\end{frame}

\begin{frame}{Limitations and Future Work}
  \textbf{Limitations:}
  \begin{itemize}
    \item Missing secondary types for ~50\% of entries
    \item No mega evolutions or alternate forms
    \item Static analysis (no battle outcomes)
  \end{itemize}
  
  \vspace{0.5cm}
  \textbf{Future Directions:}
  \begin{itemize}
    \item Incorporate move sets and abilities
    \item Model competitive viability
    \item Time-series analysis of meta shifts
    \item Network analysis of type relationships
  \end{itemize}
\end{frame}
